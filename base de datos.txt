create database robotech;
use robotech;

create table usuario(
ci int(8) not null,
nombre varchar(12) not null,
apellido varchar(12) not null,
correo varchar(30) not null,
contraseña varchar(10) not null,
permiso int(1) not null,
primary  key(ci)
);

create table administrador(
ci int(8) not null,
primary key(ci),
foreign key(ci) references usuario(ci)
);

create table vendedor(
ci int(8) not null,
primary key(ci),
foreign key(ci) references usuario(ci)
);
create table cliente(
ci int(8) not null,
cod_postal int(5) not null,
primary key(ci),
foreign key(ci) references usuario(ci)
);

create table dir_cli(
ci int(8) not null,
ciudad varchar(20) not null,
calle varchar(50) not null,
numero varchar(10) not null,
primary key(ci),
foreign key(ci) references cliente(ci)
);

create table tel_cli(
ci int(8) not null,
telefono int(9) not null,
primary key(ci, telefono),
foreign key(ci) references cliente(ci)
);

create table categoria(
id int(4) not null auto_increment,
nombre varchar(50) not null,
primary key(id)
);
create table iva(
id int(1) not null auto_increment,
tipo varchar(10) not null,
impuesto decimal(3,2) not null,
primary key(id)
);

create table proveedor(
id int(4) not null auto_increment,
nombre varchar(60) not null,
correo varchar(30) not null,
primary key(id)
);

create table dir_prov(
id int(8) not null,
ciudad varchar(20) not null,
calle varchar(50) not null,
numero varchar(10) not null,
primary key(id),
foreign key(id) references proveedor(id)
);

create table tel_prov(
id int(8) not null,
telefono int(9) not null,
primary key(id, telefono),
foreign key(id) references proveedor(id)
);

create table producto(
id int not null auto_increment,
id_categoria int(4) not null,
id_iva int(1) not null,
id_proveedor int(4) not null,
nombre varchar(80) not null,
descripcion text not null,
precio int not null,
stock int not null,
estado enum('activo', 'inactivo', 'destacado'),
primary key(id),
foreign key(id_categoria) references categoria(id),
foreign key(id_iva) references iva(id),
foreign key(id_proveedor) references proveedor(id)
);

create table metodo_pago(
id int(1) not null auto_increment,
tipo varchar(20) not null,
descuento decimal(3,2) not null,
primary key(id)    
);

create table agencia(
id int(1) not null auto_increment,
nombre varchar(20) not null,
costo int not null,
primary key(id)
);

create table envio(
id int not null auto_increment,
id_agencia int(1) not null,
despacho date not null,
primary key(id),
foreign key(id_agencia) references agencia(id)
);

create table pedido(
id int not null auto_increment,
ci_cliente int(8) not null,
ci_vendedor int(8) not null,
id_envio int not null,
id_pago int(1) not null,
fecha date not null,
primary key(id),
foreign key(ci_cliente) references cliente(ci),
foreign key(ci_vendedor) references vendedor(ci),
foreign key(id_envio) references envio(id),
foreign key(id_pago) references metodo_pago(id)
);

create table lista_productos(
id int not null auto_increment,
id_pedido int not null,
id_producto int not null,
cantidad int not null,
primary key(id, id_pedido),
foreign key(id_pedido) references pedido(id),
foreign key(id_producto) references producto(id)
);

create table garantia(
id int not null auto_increment,
detalles text not null,
expiracion date not null,
primary key(id)
);

create table factura(
id int not null auto_increment,
id_pedido int not null,
id_garantia int not null,
fecha date not null,
primary key(id),
foreign key(id_pedido) references pedido(id),
foreign key(id_garantia) references garantia(id)
);

create table reclamo(
id int not null auto_increment,
ci_cliente int(8) not null,
id_factura int not null,
detalles text not null,
fecha date not null,
primary key(id),
foreign key(ci_cliente) references cliente(ci),
foreign key(id_factura) references factura(id)
);
create table imagenes(
id_producto int not null,
primera longblob not null,
primeramin longblob not null,
segunda longblob not null,
segundamin longblob not null,
tercera longblob not null,
terceramin longblob not null,
primary key(id_producto),
foreign key(id_producto) references producto(id)
);
